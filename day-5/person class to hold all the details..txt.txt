// A class to represent a person
public class Person {
  // Class properties
  private String name; // The name of the person
  private int age; // The age of the person
  private String gender; // The gender of the person
  private String address; // The address of the person

  // A constructor that takes the name, age, gender, and address as arguments and sets the class properties accordingly
  public Person(String name, int age, String gender, String address) {
    this.name = name; // Assign the name argument to the name property
    this.age = age; // Assign the age argument to the age property
    this.gender = gender; // Assign the gender argument to the gender property
    this.address = address; // Assign the address argument to the address property
  }

  // Other methods and behaviors of the class
}
